function isPrime(num) {
    const sqrt = Math.floor(Math.sqrt(num));
    
    if(num <= 1) return false;
    
    for (let i = 2; i <= sqrt; i += 1) {
        if (num % i === 0) return false;
    }    
    return true;
}
