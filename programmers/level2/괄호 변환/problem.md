function solution(p) {
  const isCollect = (p) => {
    let count = 0;
    for (let target of p) {
      target === '(' ? count++ : count--;
      if (count < 0) return false;
    }
    return true;
  };

  const reverse = (str) => {
    return [...str.slice(1, str.length - 1)].map((c) => (c === '(' ? ')' : '(')).join('');
  };

  const check = (p) => {
    if (p.length <= 0) return '';

    let [lc, rc] = [0, 0];
    for (const c of p) {
      c === '(' ? lc++ : rc++;
      if (lc === rc) break;
    }

    const balance = lc + rc;
    const [u, v] = [p.slice(0, balance), p.slice(balance)];

    if (isCollect(u)) return u + check(v);
    else return '(' + check(v) + ')' + reverse(u);
  };

  return check(p);
}

function reverse(str) {
  return str
    .slice(1, str.length - 1)
    .split('')
    .map((c) => (c === '(' ? ')' : '('))

function solution(p) {
  if (p.length < 1) return '';

  let balance = 0;
  let pivot = 0;
  do {
    balance += p[pivot++] === '(' ? 1 : -1;
  } while (balance !== 0);

  const u = p.slice(0, pivot);
  const v = solution(p.slice(pivot, p.length));

  if (u[0] === '(' && u[u.length - 1] == ')') return u + v;
  else return '(' + v + ')' + reverse(u);
}
